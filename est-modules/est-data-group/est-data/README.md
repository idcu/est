        Product p4 = new Product();
        p4.setName("iPad");
        p4.setPrice(3999.0);
        p4.setStock(0);
        repo.save(p4);
        
        Product p5 = new Product();
        p5.setName("AirPods Pro");
        p5.setPrice(1999.0);
        p5.setStock(200);

        System.out.println("浠锋牸 >= 10000锛? + products.where(p -> p.getPrice() >= 10000).count());

            .map(p -> {
                p.setPrice(p.getPrice() * 1.1);
                return p;
            })
            .forEach(p -> {
                repo.save(p);
                System.out.println("鏇存柊鍚庯細" + p);
            });
    }
}
