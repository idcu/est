TypeScript / JavaScript SDK

                            

                                
安装

                                
npm install @est/framework-sdk
# 或
yarn add @est/framework-sdk

                            

                            

                                
基础使用

                                
import { EstClient } from '@est/framework-sdk';

// 初始化客户端
const client = new EstClient({
  apiUrl: 'http://localhost:8080',
  apiKey: 'your-api-key'
});

// 发送请求
const response = await client.request('/api/hello');
console.log(response);

                            

                            

                                
插件市场

                                
import { PluginMarketplaceClient } from '@est/framework-sdk';

const marketplace = new PluginMarketplaceClient();

// 搜索插件
const plugins = await marketplace.searchPlugins('database');
console.log(plugins);

// 安装插件
await marketplace.installPlugin('database-plugin', '1.0.0');

Go SDK

                            

                                
安装

                                
go get github.com/idcu/est-sdk-go

                            

                            

                                
基础使用

                                
package main

import (
    "fmt"
    "github.com/idcu/est-sdk-go"
)

func main() {
    // 初始化客户端
    client := est.NewEstClient(&est.ClientConfig{
        APIUrl: "http://localhost:8080",
        APIKey: "your-api-key",
    })
    
    // 发送请求
    response, err := client.Request("/api/hello")
    if err != nil {
        panic(err)
    }
    
    fmt.Println(response)
}

                            

                            

                                
插件市场

                                
package main

import (
    "fmt"
    "github.com/idcu/est-sdk-go"
)

func main() {
    marketplace := est.NewPluginMarketplaceClient()
    
    // 搜索插件
    plugins, err := marketplace.SearchPlugins("database")
    if err != nil {
        panic(err)
    }
    
    fmt.Println(plugins)
    
    // 安装插件
    err = marketplace.InstallPlugin("database-plugin", "1.0.0")
    if err != nil {
        panic(err)
    }
}

                            

                        

                        
                        

                            
Kotlin DSL

                            

                                
协程支持

                                
import ltd.idcu.est.kotlin.coroutines.*
import kotlinx.coroutines.*

fun main() = runBlocking {
    val scope = EstCoroutineScope()
    
    scope.launch {
        println("Running in EST coroutine scope")
        delay(1000)
        println("Done!")
    }
}

                            

                            

                                
EST DSL

                                
import ltd.idcu.est.kotlin.dsl.*

fun main() {
    val app = estApplication {
        name = "My EST App"
        version = "1.0.0"
        
        modules {
            +"est-core"
            +"est-web"
            +"est-plugin"
        }
        
        config {
            "server.port" to 8080
            "app.name" to "My App"
        }
    }
    
    app.start()
}

                            

                            

                                
Flow 支持

                                
import ltd.idcu.est.kotlin.flow.*
import kotlinx.coroutines.*

fun main() = runBlocking {
    val flow = estFlow {
        for (i in 1..5) {
            emit(i)
            delay(100)
        }
    }
    
    flow.collect { value ->
        println("Received: $value")
    }
}
